function showCalendar(sImg,bOpenBound,sFld1,sFld2,sCallback)
{
	var fld1,fld2;
	var cf = $("CalFrame");
	var wcf = window.frames.CalFrame;
	var oImg = $(sImg);
	if(!oImg)
	{
		alert("控制对象不存在！");
		return;
	}
	if(!sFld1)
	{
		alert("输入控件未指定！");
		return;
	}
	fld1 = $(sFld1);
	if(!fld1)
	{
		alert("输入控件不存在！");
		return;
	}
	if(sFld2)
	{
		fld2 = $(sFld2);
		if(!fld2)
		{
			alert("参考控件不存在！");
			return;
		}
		if(fld2.tagName != "INPUT" || fld2.type != "text")
		{
			alert("参考控件类型错误！");
			return;
		}
	}
	if(cf.style.display == "block")
	{
		cf.style.display = "none";
		return;
	}
	var eT = 0,eL = 0,p = oImg;
	var sT = document.body.scrollTop,sL = document.body.scrollLeft;
	var eH = oImg.height,eW = oImg.width;
	while(p &amp;&amp; p.tagName != "BODY")
	{
		eT += p.offsetTop;
		eL += p.offsetLeft;
		p = p.offsetParent;
	}
	cf.style.top = (document.body.clientHeight-(eT-sT)-eH&gt;=cf.height)?eT+eH:eT-cf.height;
	cf.style.left = (document.body.clientWidth-(eL-sL)&gt;=cf.width)?eL:eL+eW-cf.width;
	cf.style.display = "block";
	wcf.openbound = bOpenBound;
	wcf.fld1 = fld1;
	wcf.fld2 = fld2;
	wcf.callback = sCallback;
}

function historyGoTo(str)//检索结果直接跳转到指定的页码

	}

function RelatSearch(str1,str2)
{
	var strQuery = "";
	var sArry = str2.split(" ");
	for(var i = 0 ; i &lt; sArry.length;i++)
	{
		if(trim(sArry[i]) == "invention_view")
		{
			continue;
		}else if(trim(sArry[i]) == "utiltymodel_view")
		{
			continue;
		}else if(trim(sArry[i]) == "face_view")
		{
			continue;
		}else if(trim(sArry[i]) == "granted_view")
		{
			continue;
		}else
		{
			if(strQuery.indexOf(" " + trim(sArry[i])) &lt; 0)
			{
				strQuery = strQuery + " " + trim(sArry[i]);
			}
		}
	}
	strQuery = trim(strQuery);
	if(str1 == "google")
	{
		window.open('http://www.google.com/search?hl=zh-CN&amp;ie=GB2312&amp;q=' + strQuery);
	}
	if(str1 == "baidu")
	{
		window.open('http://www.baidu.com/baidu?tn=kkchat&amp;word=' + strQuery);
	}
	if(str1 == "scholargoogle")
	{
		window.open('http://www.scholar.google.com/scholar?hl=zh-CN&amp;q=' + encodeURIComponent(strQuery) + "&amp;btnG=%E6%90%9C%E7%B4%A2&amp;lr=");
	}
}
