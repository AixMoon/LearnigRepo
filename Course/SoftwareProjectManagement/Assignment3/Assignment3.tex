\section{Q1}

\begin{align*}
t_A &= \frac{a + 4m + b}{6}  = \frac{8 + 40 + 24}{6} = 12 \\
t_B &= \frac{a + 4m + b}{6} = \frac{1 + 20 + 9}{6} = 5 \\
t_e &= t_A + t_B = 17 \text{ days.} \\
s_a &= \frac{b-a}{6} = \frac{8}{3} \\
s_b &= \frac{b-a}{6} = \frac{4}{3} \\
S &= \sqrt{\sum_{i \in \{a, b\}}s_i^2} \approx 3 \text{ days.} \\
Z &= \frac{T-t_e}{S} = 1 \\
P & = \Phi(1) = 0.843\\
\end{align*}
